155  template <typename Type> detail::FormWidget<Type> *
156  add_variable(const std::string &label, const std::function<void(const Type &)> &setter,
157  const std::function<Type()> &getter, bool editable = true) {
158  Label *label_w = new Label(m_window, label, m_label_font_name, m_label_font_size);
159  auto widget = new detail::FormWidget<Type>(m_window);
160  auto refresh = [widget, getter] {
161  Type value = getter(), current = widget->value();
162  if (value != current)
163  widget->set_value(value);
164  };
165  refresh();
166  widget->set_callback(setter);
167  widget->set_editable(editable);
168  widget->set_font_size(m_widget_font_size);
169  Vector2i fs = widget->fixed_size();
170  widget->set_fixed_size(Vector2i(fs.x() != 0 ? fs.x() : m_fixed_size.x(),
171  fs.y() != 0 ? fs.y() : m_fixed_size.y()));
172  m_refresh_callbacks.push_back(refresh);
173  if (m_layout->row_count() > 0)
174  m_layout->append_row(m_variable_spacing);
175  m_layout->append_row(0);
176  m_layout->set_anchor(label_w, AdvancedGridLayout::Anchor(1, m_layout->row_count()-1));
177  m_layout->set_anchor(widget, AdvancedGridLayout::Anchor(3, m_layout->row_count()-1));
178  return widget;
179  }

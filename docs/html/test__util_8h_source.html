92  CommandLineArgs(int argc, char **argv) :
93  keys(10),
94  values(10)
95  {
96  using namespace std;
97 
98  // Initialize mersenne generator
99  unsigned int mersenne_init[4]= {0x123, 0x234, 0x345, 0x456};
100  mersenne::init_by_array(mersenne_init, 4);
101 
102  for (int i = 1; i < argc; i++)
103  {
104  string arg = argv[i];
105 
106  if ((arg[0] != '-') || (arg[1] != '-'))
107  {
108  args.push_back(arg);
109  continue;
110  }
111 
112  string::size_type pos;
113  string key, val;
114  if ((pos = arg.find('=')) == string::npos) {
115  key = string(arg, 2, arg.length() - 2);
116  val = "";
117  } else {
118  key = string(arg, 2, pos - 2);
119  val = string(arg, pos + 1, arg.length() - 1);
120  }
121 
122  keys.push_back(key);
123  values.push_back(val);
124  }
125  }

189  template <typename T>
190  void GetCmdLineArguments(const char *arg_name, std::vector<T> &vals)
191  {
192  using namespace std;
193 
194  if (CheckCmdLineFlag(arg_name))
195  {
196  // Clear any default values
197  vals.clear();
198 
199  // Recover from multi-value string
200  for (int i = 0; i < keys.size(); ++i)
201  {
202  if (keys[i] == string(arg_name))
203  {
204  string val_string(values[i]);
205  istringstream str_stream(val_string);
206  string::size_type old_pos = 0;
207  string::size_type new_pos = 0;
208 
209  // Iterate comma-separated values
210  T val;
211  while ((new_pos = val_string.find(',', old_pos)) != string::npos)
212  {
213  if (new_pos != old_pos)
214  {
215  str_stream.width(new_pos - old_pos);
216  str_stream >> val;
217  vals.push_back(val);
218  }
219 
220  // skip over comma
221  str_stream.ignore(1);
222  old_pos = new_pos + 1;
223  }
224 
225  // Read last value
226  str_stream >> val;
227  vals.push_back(val);
228  }
229  }
230  }
231  }

245  cudaError_t DeviceInit(int dev = -1)
246  {
247  cudaError_t error = cudaSuccess;
248 
249  do
250  {
251  int deviceCount;
252  error = CubDebug(cudaGetDeviceCount(&deviceCount));
253  if (error) break;
254 
255  if (deviceCount == 0) {
256  fprintf(stderr, "No devices supporting CUDA.\n");
257  exit(1);
258  }
259  if (dev < 0)
260  {
261  GetCmdLineArgument("device", dev);
262  }
263  if ((dev > deviceCount - 1) || (dev < 0))
264  {
265  dev = 0;
266  }
267 
268  error = CubDebug(cudaSetDevice(dev));
269  if (error) break;
270 
271  CubDebugExit(cudaMemGetInfo(&device_free_physmem, &device_total_physmem));
272 
273  int ptx_version;
274  error = CubDebug(cub::PtxVersion(ptx_version));
275  if (error) break;
276 
277  error = CubDebug(cudaGetDeviceProperties(&deviceProp, dev));
278  if (error) break;
279 
280  if (deviceProp.major < 1) {
281  fprintf(stderr, "Device does not support CUDA.\n");
282  exit(1);
283  }
284 
285  device_giga_bandwidth = float(deviceProp.memoryBusWidth) * deviceProp.memoryClockRate * 2 / 8 / 1000 / 1000;
286 
287  if (!CheckCmdLineFlag("quiet"))
288  {
289  printf(
290  "Using device %d: %s (PTX version %d, SM%d, %d SMs, "
291  "%lld free / %lld total MB physmem, "
292  "%.3f GB/s @ %d kHz mem clock, ECC %s)\n",
293  dev,
294  deviceProp.name,
295  ptx_version,
296  deviceProp.major * 100 + deviceProp.minor * 10,
297  deviceProp.multiProcessorCount,
298  (unsigned long long) device_free_physmem / 1024 / 1024,
299  (unsigned long long) device_total_physmem / 1024 / 1024,
300  device_giga_bandwidth,
301  deviceProp.memoryClockRate,
302  (deviceProp.ECCEnabled) ? "on" : "off");
303  fflush(stdout);
304  }
305 
306  } while (0);
307 
308  return error;
309  }

422 template <typename K>
423 void RandomBits(
424  K &key,
425  int entropy_reduction = 0,
426  int begin_bit = 0,
427  int end_bit = sizeof(K) * 8)
428 {
429  const int NUM_BYTES = sizeof(K);
430  const int WORD_BYTES = sizeof(unsigned int);
431  const int NUM_WORDS = (NUM_BYTES + WORD_BYTES - 1) / WORD_BYTES;
432 
433  unsigned int word_buff[NUM_WORDS];
434 
435  if (entropy_reduction == -1)
436  {
437  memset((void *) &key, 0, sizeof(key));
438  return;
439  }
440 
441  if (end_bit < 0)
442  end_bit = sizeof(K) * 8;
443 
444  while (true)
445  {
446  // Generate random word_buff
447  for (int j = 0; j < NUM_WORDS; j++)
448  {
449  int current_bit = j * WORD_BYTES * 8;
450 
451  unsigned int word = 0xffffffff;
452  word &= 0xffffffff << CUB_MAX(0, begin_bit - current_bit);
453  word &= 0xffffffff >> CUB_MAX(0, (current_bit + (WORD_BYTES * 8)) - end_bit);
454 
455  for (int i = 0; i <= entropy_reduction; i++)
456  {
457  // Grab some of the higher bits from rand (better entropy, supposedly)
458  word &= mersenne::genrand_int32();
459  g_num_rand_samples++;
460  }
461 
462  word_buff[j] = word;
463  }
464 
465  memcpy(&key, word_buff, sizeof(K));
466 
467  K copy = key;
468  if (!IsNaN(copy))
469  break; // avoids NaNs when generating random floating point numbers
470  }
471 }

474 template <typename T>
475 T RandomValue(T max)
476 {
477  unsigned int bits;
478  unsigned int max_int = (unsigned int) -1;
479  do {
480  RandomBits(bits);
481  } while (bits == max_int);
482 
483  return (T) ((double(bits) / double(max_int)) * double(max));
484 }

512 enum GenMode
513 {
514  UNIFORM, // Assign to '2', regardless of integer seed
515  INTEGER_SEED, // Assign to integer seed
516  RANDOM, // Assign to random, regardless of integer seed
517  RANDOM_BIT, // Assign to randomly chosen 0 or 1, regardless of integer seed
518 };

523 template <typename T>
524 __host__ __device__ __forceinline__ void InitValue(GenMode gen_mode, T &value, int index = 0)
525 {
526  switch (gen_mode)
527  {
528 #if (CUB_PTX_ARCH == 0)
529  case RANDOM:
530  RandomBits(value);
531  break;
532  case RANDOM_BIT:
533  char c;
534  RandomBits(c, 0, 0, 1);
535  value = (c > 0) ? (T) 1 : (T) -1;
536  break;
537 #endif
538  case UNIFORM:
539  value = 2;
540  break;
541  case INTEGER_SEED:
542  default:
543  value = (T) index;
544  break;
545  }
546 }
547 
548 
552 __host__ __device__ __forceinline__ void InitValue(GenMode gen_mode, bool &value, int index = 0)
553 {
554  switch (gen_mode)
555  {
556 #if (CUB_PTX_ARCH == 0)
557  case RANDOM:
558  case RANDOM_BIT:
559  char c;
560  RandomBits(c, 0, 0, 1);
561  value = (c > 0);
562  break;
563 #endif
564  case UNIFORM:
565  value = true;
566  break;
567  case INTEGER_SEED:
568  default:
569  value = (index > 0);
570  break;
571  }
572 }

187  template <typename Stream, typename Array, size_t N, typename... Indices>
188  Stream &print(Stream &os, const Array &a, bool abbrev,
189  const std::array<size_t, N> &size, Indices... indices) {
190  ENOKI_MARK_USED(size);
191  ENOKI_MARK_USED(abbrev);
192  if constexpr (sizeof...(Indices) == N) {
193  os << a.derived().coeff(indices...);
194  } else {
195  constexpr size_t k = N - sizeof...(Indices) - 1;
196  os << "[";
197  for (size_t i = 0; i < size[k]; ++i) {
198  if constexpr (is_dynamic_array_v<Array>) {
199  if (size[k] > 20 && i == 5 && abbrev) {
200  if (k > 0) {
201  os << ".. " << size[k] - 10 << " skipped ..,\n";
202  for (size_t j = 0; j <= sizeof...(Indices); ++j)
203  os << " ";
204  } else {
205  os << ".. " << size[k] - 10 << " skipped .., ";
206  }
207  i = size[k] - 6;
208  continue;
209  }
210  }
211  print(os, a, abbrev, size, i, indices...);
212  if (i + 1 < size[k]) {
213  if constexpr (k == 0) {
214  os << ", ";
215  } else {
216  os << ",\n";
217  for (size_t j = 0; j <= sizeof...(Indices); ++j)
218  os << " ";
219  }
220  }
221  }
222  os << "]";
223  }
224  return os;
225  }

216  bool motion(Vector2i pos) {
217  if (!m_active)
218  return false;
219 
220  /* Based on the rotation controller form AntTweakBar */
221  float inv_min_dim = 1.0f / hmin(m_size);
222  float w = (float) m_size.x(), h = (float) m_size.y();
223 
224  float ox = (m_speed_factor * (2*m_last_pos.x() - w) + w) - w - 1.0f;
225  float tx = (m_speed_factor * (2*pos.x() - w) + w) - w - 1.0f;
226  float oy = (m_speed_factor * (h - 2*m_last_pos.y()) + h) - h - 1.0f;
227  float ty = (m_speed_factor * (h - 2*pos.y()) + h) - h - 1.0f;
228 
229  ox *= inv_min_dim; oy *= inv_min_dim;
230  tx *= inv_min_dim; ty *= inv_min_dim;
231 
232  Vector3f v0(ox, oy, 1.0f), v1(tx, ty, 1.0f);
233  if (enoki::squared_norm(v0) > 1e-4f && enoki::squared_norm(v1) > 1e-4f) {
234  v0 = enoki::normalize(v0);
235  v1 = enoki::normalize(v1);
236  Vector3f axis = enoki::cross(v0, v1);
237  float sa = enoki::norm(axis),
238  ca = enoki::dot(v0, v1),
239  angle = std::atan2(sa, ca);
240  if (tx*tx + ty*ty > 1.0f)
241  angle *= 1.0f + 0.2f * (std::sqrt(tx*tx + ty*ty) - 1.0f);
242  m_incr = enoki::rotate<Quaternion4f>(enoki::normalize(axis), angle);
243  if (!std::isfinite(enoki::abs(m_incr)))
244  m_incr = enoki::identity<Quaternion4f>();
245  }
246  return true;
247  }

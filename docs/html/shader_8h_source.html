93  template <typename Array> void set_uniform(const std::string &name,
94  const Array &value) {
95  std::array<size_t, 3> shape = { 1, 1, 1 };
96  size_t ndim;
97  const void *data;
98 
99  if constexpr (enoki::array_depth_v<Array> == 0) {
100  data = &value;
101  ndim = 0;
102  } else if constexpr (enoki::array_depth_v<Array> == 1) {
103  data = value.data();
104  shape[0] = value.size();
105  ndim = 1;
106  } else if constexpr (enoki::array_depth_v<Array> == 2) {
107  data = value.data();
108  shape[0] = value.size();
109  shape[1] = value[0].size();
110  ndim = 2;
111  } else if constexpr (enoki::array_depth_v<Array> == 3) {
112  data = value.data();
113  shape[0] = value.size();
114  shape[1] = value[0].size();
115  shape[2] = value[0][0].size();
116  ndim = 3;
117  } else {
118  throw std::runtime_error("Shader::set_uniform(): invalid input array dimension!");
119  }
120  set_buffer(name, enoki::enoki_type_v<enoki::scalar_t<Array>>, ndim, shape, data);
121  }

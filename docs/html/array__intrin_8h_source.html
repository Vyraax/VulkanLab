165 #if defined(ENOKI_X86_SSE42)
166 inline void set_flush_denormals(bool value) {
168  _MM_SET_FLUSH_ZERO_MODE(value ? _MM_FLUSH_ZERO_ON : _MM_FLUSH_ZERO_OFF);
169  _MM_SET_DENORMALS_ZERO_MODE(value ? _MM_DENORMALS_ZERO_ON : _MM_DENORMALS_ZERO_OFF);
170 }
171 
172 inline bool flush_denormals() {
173  return _MM_GET_FLUSH_ZERO_MODE() == _MM_FLUSH_ZERO_ON;
174 }
175 
176 #else
177 inline void set_flush_denormals(bool) { }
178 inline bool flush_denormals() { return false; }
179 #endif
180 
181 struct scoped_flush_denormals {
182 public:
183  scoped_flush_denormals(bool value) {
184  m_old_value = flush_denormals();
185  set_flush_denormals(value);
186 
187  }
188 
189  ~scoped_flush_denormals() {
190  set_flush_denormals(m_old_value);
191  }
192 private:
193  bool m_old_value;
194 };

244 ENOKI_INLINE __m256i mm256_cvtepi32_epi64(__m128i x) {
245 #if defined(ENOKI_X86_AVX2)
246  return _mm256_cvtepi32_epi64(x);
247 #else
248  /* This version is only suitable for mask conversions */
249  __m128i xl = _mm_shuffle_epi32(x, _MM_SHUFFLE(1, 1, 0, 0));

255 ENOKI_INLINE __m128i mm256_cvtepi64_epi32(__m256i x) {
256 #if defined(ENOKI_X86_AVX512VL)
257  return _mm256_cvtepi64_epi32(x);
258 #else
259  __m128i x0 = _mm256_castsi256_si128(x);
260  __m128i x1 = _mm256_extractf128_si256(x, 1);
261  return _mm_castps_si128(_mm_shuffle_ps(
262  _mm_castsi128_ps(x0), _mm_castsi128_ps(x1), _MM_SHUFFLE(2, 0, 2, 0)));
263 #endif
264 }

266 ENOKI_INLINE __m256i mm512_cvtepi64_epi32(__m128i x0, __m128i x1, __m128i x2, __m128i x3) {
267  __m128i y0 = _mm_castps_si128(_mm_shuffle_ps(
268  _mm_castsi128_ps(x0), _mm_castsi128_ps(x1), _MM_SHUFFLE(2, 0, 2, 0)));
269  __m128i y1 = _mm_castps_si128(_mm_shuffle_ps(
270  _mm_castsi128_ps(x2), _mm_castsi128_ps(x3), _MM_SHUFFLE(2, 0, 2, 0)));
271  return detail::concat(y0, y1);
272 }

37 template <typename Matrix, typename Vector3, enable_if_t<Matrix::IsMatrix && Matrix::Size == 4> = 0>
38 ENOKI_INLINE Matrix rotate(const Vector3 &axis, const entry_t<Matrix> &angle) {
39  using Value = entry_t<Matrix>;
40  using Vector4 = column_t<Matrix>;
41 
42  auto [sin_theta, cos_theta] = sincos(angle);
43  Value cos_theta_m = 1.f - cos_theta;
44 
45  auto shuf1 = shuffle<1, 2, 0>(axis),
46  shuf2 = shuffle<2, 0, 1>(axis),
47  tmp0 = fmadd(axis * axis, cos_theta_m, cos_theta),
48  tmp1 = fmadd(axis * shuf1, cos_theta_m, shuf2 * sin_theta),
49  tmp2 = fmsub(axis * shuf2, cos_theta_m, shuf1 * sin_theta);
50 
51  return Matrix(
52  Vector4(tmp0.x(), tmp1.x(), tmp2.x(), 0.f),
53  Vector4(tmp2.y(), tmp0.y(), tmp1.y(), 0.f),
54  Vector4(tmp1.z(), tmp2.z(), tmp0.z(), 0.f),
55  Vector4(0.f, 0.f, 0.f, 1.f)
56  );
57 }

59 template <typename Matrix>
60 ENOKI_INLINE Matrix perspective(const entry_t<Matrix> &fov,
61  const entry_t<Matrix> &near_,
62  const entry_t<Matrix> &far_,
63  const entry_t<Matrix> &aspect = 1.f) {
64  static_assert(Matrix::Size == 4, "Matrix::perspective(): implementation assumes 4x4 matrix output");
65 
66  auto recip = rcp(near_ - far_);
67  auto c = cot(.5f * fov);
68 
69  Matrix trafo = diag<Matrix>(
70  column_t<Matrix>(c / aspect, c, (near_ + far_) * recip, 0.f));
71 
72  trafo(2, 3) = 2.f * near_ * far_ * recip;
73  trafo(3, 2) = -1.f;
74 
75  return trafo;
76 }

78 template <typename Matrix>
79 ENOKI_INLINE Matrix frustum(const entry_t<Matrix> &left,
80  const entry_t<Matrix> &right,
81  const entry_t<Matrix> &bottom,
82  const entry_t<Matrix> &top,
83  const entry_t<Matrix> &near_,
84  const entry_t<Matrix> &far_) {
85  static_assert(Matrix::Size == 4, "Matrix::frustum(): implementation assumes 4x4 matrix output");
86 
87  auto rl = rcp(right - left),
88  tb = rcp(top - bottom),
89  fn = rcp(far_ - near_);
90 
91  Matrix trafo = zero<Matrix>();
92  trafo(0, 0) = (2.f * near_) * rl;
93  trafo(1, 1) = (2.f * near_) * tb;
94  trafo(0, 2) = (right + left) * rl;
95  trafo(1, 2) = (top + bottom) * tb;
96  trafo(2, 2) = -(far_ + near_) * fn;
97  trafo(3, 2) = -1.f;
98  trafo(2, 3) = -2.f * far_ * near_ * fn;
99 
100  return trafo;
101 }

103 template <typename Matrix>
104 ENOKI_INLINE Matrix ortho(const entry_t<Matrix> &left,
105  const entry_t<Matrix> &right,
106  const entry_t<Matrix> &bottom,
107  const entry_t<Matrix> &top,
108  const entry_t<Matrix> &near_,
109  const entry_t<Matrix> &far_) {
110  static_assert(Matrix::Size == 4, "Matrix::ortho(): implementation assumes 4x4 matrix output");
111 
112  auto rl = rcp(right - left),
113  tb = rcp(top - bottom),
114  fn = rcp(far_ - near_);
115 
116  Matrix trafo = zero<Matrix>();
117 
118  trafo(0, 0) = 2.f * rl;
119  trafo(1, 1) = 2.f * tb;
120  trafo(2, 2) = -2.f * fn;
121  trafo(3, 3) = 1.f;
122  trafo(0, 3) = -(right + left) * rl;
123  trafo(1, 3) = -(top + bottom) * tb;
124  trafo(2, 3) = -(far_ + near_) * fn;
125 
126  return trafo;
127 }

129 template <typename Matrix, typename Point, typename Vector>
130 Matrix look_at(const Point &origin, const Point &target, const Vector &up) {
131  static_assert(Matrix::Size == 4, "Matrix::look_at(): implementation "
132  "assumes 4x4 matrix output");
133 
134  auto dir = normalize(target - origin);
135  auto left = normalize(cross(dir, up));
136  auto new_up = cross(left, dir);
137  using Scalar = scalar_t<Matrix>;
138 
139  return Matrix(
140  concat(left, Scalar(0)),
141  concat(new_up, Scalar(0)),
142  concat(-dir, Scalar(0)),
143  column_t<Matrix>(
144  -dot(left, origin),
145  -dot(new_up, origin),
146  dot(dir, origin),
147  1.f
148  )
149  );
150 }

152 template <typename T, bool Approx,
153  typename E = expr_t<T>,
154  typename Matrix3 = Matrix<E, 3, Approx>,
155  typename Vector3 = Array<E, 3, Approx>,
156  typename Quat = Quaternion<E, Approx>>
157 std::tuple<Matrix3, Quat, Vector3> transform_decompose(const Matrix<T, 4, Approx> &A) {
158  Matrix3 A_sub(A), Q, P;
159  std::tie(Q, P) = polar_decomp(A_sub);
160 
161  if (ENOKI_UNLIKELY(any(enoki::isnan(Q(0, 0)))))
162  Q = identity<Matrix3>();
163 
164  auto sign_q = det(Q);
165  Q = mulsign(Array<Vector3, 3>(Q), sign_q);
166  P = mulsign(Array<Vector3, 3>(P), sign_q);
167 
168  return std::make_tuple(
169  P,
170  matrix_to_quat(Q),
171  head<3>(A.col(3))
172  );
173 }

39 template <bool UseIntrinsic = false>
40 std::pair<uint32_t, uint32_t> div_wide(uint32_t u1, uint32_t u0, uint32_t v) {
41 #if defined(__GNUC__) && (defined(ENOKI_X86_32) || defined(ENOKI_X86_64))
42  if constexpr (UseIntrinsic) {
43  uint32_t res, rem;
44  __asm__("divl %[v]"
45  : "=a"(res), "=d"(rem)
46  : [v] "r"(v), "a"(u0), "d"(u1));
47  return { res, rem };
48  }
49 #endif
50 
51  uint64_t u = (((uint64_t) u1) << 32) | u0;
52 
53  return { (uint32_t) (u / v),
54  (uint32_t) (u % v) };
55 }

57 template <bool UseIntrinsic = false>
58 std::pair<uint64_t, uint64_t> div_wide(uint64_t u1, uint64_t u0, uint64_t d) {
59 #if defined(__GNUC__) && defined(ENOKI_X86_64)
60  if constexpr (UseIntrinsic) {
61  uint64_t res, rem;
62  __asm__("divq %[v]"
63  : "=a"(res), "=d"(rem)
64  : [v]"r"(d), "a"(u0), "d"(u1));
65  return { res, rem };
66  }
67 #endif
68 
69 #if defined(__SIZEOF_INT128__)
70  __uint128_t n = (((__uint128_t) u1) << 64) | u0;
71  return {
72  (uint64_t) (n / d),
73  (uint64_t) (n % d)
74  };
75 #else
76  // Code taken from Hacker's Delight:
77  // http://www.hackersdelight.org/HDcode/divlu.c.
78  // License permits inclusion here per:
79  // http://www.hackersdelight.org/permissions.htm
80 
81  const uint64_t b = (1ULL << 32); // Number base (16 bits).
82  uint64_t un1, un0, // Norm. dividend LSD's.
83  vn1, vn0, // Norm. divisor digits.
84  q1, q0, // Quotient digits.
85  un64, un21, un10, // Dividend digit pairs.
86  rhat; // A remainder.
87  int s; // Shift amount for norm.
88 
89  if (u1 >= d) // overflow
90  return { (uint64_t) -1, (uint64_t) -1 };
91 
92  // count leading zeros
93  s = (int) (63 - log2i(d)); // 0 <= s <= 63.
94  if (s > 0) {
95  d = d << s; // Normalize divisor.
96  un64 = (u1 << s) | ((u0 >> (64 - s)) & uint64_t(-s >> 31));
97  un10 = u0 << s; // Shift dividend left.
98  } else {
99  // Avoid undefined behavior.
100  un64 = u1 | u0;
101  un10 = u0;
102  }
103 
104  vn1 = d >> 32; // Break divisor up into
105  vn0 = d & 0xFFFFFFFF; // two 32-bit digits.
106 
107  un1 = un10 >> 32; // Break right half of
108  un0 = un10 & 0xFFFFFFFF; // dividend into two digits.
109 
110  q1 = un64/vn1; // Compute the first
111  rhat = un64 - q1*vn1; // quotient digit, q1.
112 
113 again1:
114  if (q1 >= b || q1*vn0 > b*rhat + un1) {
115  q1 = q1 - 1;
116  rhat = rhat + vn1;
117  if (rhat < b)
118  goto again1;
119  }
120 
121  un21 = un64*b + un1 - q1*d; // Multiply and subtract.
122 
123  q0 = un21/vn1; // Compute the second
124  rhat = un21 - q0*vn1; // quotient digit, q0.
125 
126 again2:
127  if (q0 >= b || q0 * vn0 > b * rhat + un0) {
128  q0 = q0 - 1;
129  rhat = rhat + vn1;
130  if (rhat < b)
131  goto again2;
132  }
133 
134  return {
135  q1*b + q0,
136  (un21*b + un0 - q0*d) >> s
137  };
138 #endif
139 }

151 template <typename T, bool UseIntrinsic>
152 struct divisor<T, UseIntrinsic, enable_if_t<std::is_unsigned_v<T>>> {
153  T multiplier;
154  uint8_t shift;
155 
156  divisor() = default;
157 
158  ENOKI_INLINE divisor(T d) {
159  /* Division by +/-1 is not supported by the
160  precomputation-based approach */
161  assert(d != 1);
162  shift = (uint8_t) log2i(d);
163 
164  if ((d & (d - 1)) == 0) {
165  /* Power of two */
166  multiplier = 0;
167  shift--;
168  } else {
169  /* General case */
170  auto [m, rem] =
171  detail::div_wide<UseIntrinsic>(T(1) << shift, T(0), d);
172  multiplier = m * 2 + 1;
173  assert(rem > 0 && rem < d);
174 
175  T rem2 = rem * 2;
176  if (rem2 >= d || rem2 < rem)
177  multiplier += 1;
178  }
179  }
180 
181  template <typename T2>
182  ENOKI_INLINE auto operator()(const T2 &value) const {
183  using Expr = decltype(value + value);
184  auto q = mulhi(Expr(multiplier), value);
185  auto t = sr<1>(value - q) + q;
186  return t >> shift;
187  }
188 } ENOKI_PACK;

190 template <typename T, bool UseIntrinsic>
191 struct divisor<T, UseIntrinsic, enable_if_t<std::is_signed_v<T>>> {
192  using U = std::make_unsigned_t<T>;
193 
194  T multiplier;
195  uint8_t shift;
196 
197  divisor() = default;
198 
199  ENOKI_INLINE divisor(T d) {
200  /* Division by +/-1 is not supported by the
201  precomputation-based approach */
202  assert(d != 1 && d != -1);
203 
204  U ad = d < 0 ? (U) -d : (U) d;
205  shift = (uint8_t) log2i(ad);
206 
207  if ((ad & (ad - 1)) == 0) {
208  /* Power of two */
209  multiplier = 0;
210  } else {
211  /* General case */
212  auto [m, rem] =
213  detail::div_wide<UseIntrinsic>(U(1) << (shift - 1), U(0), ad);
214  multiplier = T(m * 2 + 1);
215 
216  U rem2 = rem * 2;
217  if (rem2 >= ad || rem2 < rem)
218  multiplier += 1;
219  }
220  if (d < 0)
221  shift |= 0x80;
222  }
223 
224  template <typename T2>
225  ENOKI_INLINE auto operator()(const T2 &value) const {
226  using Expr = decltype(value + value);
227  uint8_t shift_ = shift & 0x3f;
228  Expr sign(int8_t(shift) >> 7);
229 
230  auto q = mulhi(Expr(multiplier), value) + value;
231  auto q_sign = sr<sizeof(T) * 8 - 1>(q);
232  q += q_sign & ((T(1) << shift_) - (multiplier == 0 ? 1 : 0));
233 
234  return ((q >> shift_) ^ sign) - sign;
235  }
236 } ENOKI_PACK;

262 template <typename T1, typename T2,
263  typename S1 = scalar_t<T1>, typename S2 = scalar_t<T2>,
264  enable_if_t<std::is_pointer_v<S1> || std::is_pointer_v<S2>> = 0,
265  enable_if_array_any_t<T1, T2> = 0>
266 ENOKI_INLINE auto operator-(const T1 &a1_, const T2 &a2_) {
267  using Int = std::conditional_t<sizeof(void *) == 8, int64_t, int32_t>;
268  using T1i = replace_scalar_t<T1, Int, false>;
269  using T2i = replace_scalar_t<T2, Int, false>;
270  using Ti = expr_t<T1i, T2i>;
271  using T = expr_t<T1, T2>;
272 
273  constexpr Int InstanceSize = sizeof(std::remove_pointer_t<scalar_t<T1>>),
274  LogInstanceSize = detail::clog2i(InstanceSize);
275 
276  constexpr bool PointerDiff = std::is_pointer_v<S1> &&
277  std::is_pointer_v<S2>;
278 
279  using Ret = std::conditional_t<PointerDiff, Ti, T>;
280  Ti a1 = Ti((T1i) a1_),
281  a2 = Ti((T2i) a2_);
282 
283  if constexpr (InstanceSize == 1) {
284  return Ret(a1.sub_(a2));
285  } else if constexpr ((1 << LogInstanceSize) == InstanceSize) {
286  if constexpr (PointerDiff)
287  return Ret(a1.sub_(a2).template sr_<LogInstanceSize>());
288  else
289  return Ret(a1.sub_(a2.template sl_<LogInstanceSize>()));
290  } else {
291  if constexpr (PointerDiff)
292  return Ret(a1.sub_(a2) / InstanceSize);
293  else
294  return Ret(a1.sub_(a2 * InstanceSize));
295  }
296 }

299 template <typename T1, typename T2,
300  typename S1 = scalar_t<T1>, typename S2 = scalar_t<T2>,
301  enable_if_t<std::is_pointer_v<S1> && !std::is_pointer_v<S2>> = 0,
302  enable_if_array_any_t<T1, T2> = 0>
303 ENOKI_INLINE auto operator+(const T1 &a1_, const T2 &a2_) {
304  using Int = std::conditional_t<sizeof(void *) == 8, int64_t, int32_t>;
305  using T1i = replace_scalar_t<T1, Int, false>;
306  using T2i = replace_scalar_t<T2, Int, false>;
307  using Ti = expr_t<T1i, T2i>;
308  using Ret = expr_t<T1, T2>;
309 
310  constexpr Int InstanceSize = sizeof(std::remove_pointer_t<scalar_t<T1>>),
311  LogInstanceSize = detail::clog2i(InstanceSize);
312 
313  Ti a1 = Ti((T1i) a1_),
314  a2 = Ti((T2i) a2_);
315 
316  if constexpr (InstanceSize == 1)
317  return Ret(a1.add_(a2));
318  if constexpr ((1 << LogInstanceSize) == InstanceSize)
319  return Ret(a1.add_(a2.template sl_<LogInstanceSize>()));
320  else
321  return Ret(a1.add_(a2 * InstanceSize));
322 }

312  ENOKI_INLINE Derived rcp_() const {
313  #if defined(ENOKI_X86_AVX512ER)
314  /* rel err < 2^28, use as is */
315  return _mm512_rcp28_ps(m);
316  #else
317  if constexpr (Approx_) {
318  /* Use best reciprocal approximation available on the current
319  hardware and refine */
320  __m512 r = _mm512_rcp14_ps(m); /* rel error < 2^-14 */
321 
322  /* Refine using one Newton-Raphson iteration */
323  __m512 t0 = _mm512_add_ps(r, r),
324  t1 = _mm512_mul_ps(r, m);
325 
326  r = _mm512_fnmadd_ps(t1, r, t0);
327 
328  return _mm512_fixupimm_ps(r, m,
329  _mm512_set1_epi32(0x0087A622), 0);
330  } else {
331  return (Scalar) 1 / derived();
332  }
333  #endif
334  }

336  ENOKI_INLINE Derived rsqrt_() const {
337  #if defined(ENOKI_X86_AVX512ER)
338  /* rel err < 2^28, use as is */
339  return _mm512_rsqrt28_ps(m);
340  #else
341  if constexpr (Approx_) {
342  __m512 r = _mm512_rsqrt14_ps(m); /* rel error < 2^-14 */
343 
344  /* Refine using one Newton-Raphson iteration */
345  const __m512 c0 = _mm512_set1_ps(0.5f),
346  c1 = _mm512_set1_ps(3.0f);
347 
348  __m512 t0 = _mm512_mul_ps(r, c0),
349  t1 = _mm512_mul_ps(r, m);
350 
351  r = _mm512_mul_ps(_mm512_fnmadd_ps(t1, r, c1), t0);
352 
353  return _mm512_fixupimm_ps(r, m,
354  _mm512_set1_epi32(0x0383A622), 0);
355  } else {
356  return (Scalar) 1 / sqrt(derived());
357  }

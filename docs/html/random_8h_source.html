62  void seed(const UInt64 &initstate, const UInt64 &initseq) {
63  state = zero<UInt64>();
64  inc = sl<1>(initseq) | 1u;
65  next_uint32();
66  state += initstate;
67  next_uint32();
68  }

71  ENOKI_INLINE UInt32 next_uint32() {
72  UInt64 oldstate = state;
73  state = oldstate * uint64_t(PCG32_MULT) + inc;
74  UInt32 xorshifted = UInt32(sr<27>(sr<18>(oldstate) ^ oldstate));
75  UInt32 rot_offset = UInt32(sr<59>(oldstate));
76  return ror(xorshifted, rot_offset);
77  }
78 
80  ENOKI_INLINE UInt32 next_uint32(const UInt64Mask &mask) {
81  UInt64 oldstate = state;
82  masked(state, mask) = oldstate * uint64_t(PCG32_MULT) + inc;
83  UInt32 xorshifted = UInt32(sr<27>(sr<18>(oldstate) ^ oldstate));
84  UInt32 rot_offset = UInt32(sr<59>(oldstate));
85  return ror(xorshifted, rot_offset);
86  }

89  template <typename Indices, enable_if_dynamic_t<Indices> = 0>
90  ENOKI_INLINE UInt32 next_uint32(const Indices &indices, const UInt64Mask &mask) {
91  UInt64 oldstate = gather<UInt64>(state, indices, mask),
92  inc_i = gather<UInt64>(inc, indices, mask);
93  scatter(state, oldstate * uint64_t(PCG32_MULT) + inc_i, indices, mask);
94  UInt32 xorshifted = UInt32(sr<27>(sr<18>(oldstate) ^ oldstate));
95  UInt32 rot_offset = UInt32(sr<59>(oldstate));
96  return ror(xorshifted, rot_offset);
97  }

159  UInt32 next_uint32_bounded(uint32_t bound, UInt64Mask mask = true) {
160  if constexpr (is_scalar_v<T>) {
161  ENOKI_MARK_USED(mask);
162 
163  // To avoid bias, we need to make the range of the RNG a multiple of
164  // bound, which we do by dropping output less than a threshold.
165  // A naive scheme to calculate the threshold would be to do
166  //
167  // UInt32 threshold = 0x1'0000'0000ull % bound;
168  //
169  // but 64-bit div/mod is slower than 32-bit div/mod (especially on
170  // 32-bit platforms). In essence, we do
171  //
172  // UInt32 threshold = (0x1'0000'0000ull-bound) % bound;
173  //
174  // because this version will calculate the same modulus, but the LHS
175  // value is less than 2^32.
176 
177  const UInt32 threshold = (~bound + 1u) % bound;
178 
179  // Uniformity guarantees that this loop will terminate. In practice, it
180  // should usually terminate quickly; on average (assuming all bounds are
181  // equally likely), 82.25% of the time, we can expect it to require just
182  // one iteration. In the worst case, someone passes a bound of 2^31 + 1
183  // (i.e., 2147483649), which invalidates almost 50% of the range. In
184  // practice, bounds are typically small and only a tiny amount of the range
185  // is eliminated.
186 
187  while (true) {
188  UInt32 result = next_uint32();
189 
190  if (all(result >= threshold))
191  return result % bound;
192  }
193  } else {
194  const divisor_ext<uint32_t> div(bound);
195  const UInt32 threshold = (~bound + 1u) % div;
196 
197  UInt32 result = zero<UInt32>();
198  do {
199  result[mask] = next_uint32(mask);
200 
201  /* Keep track of which SIMD lanes have already
202  finished and stops advancing the associated PRNGs */
203  mask &= result < threshold;
204  } while (any(mask));
205 
206  return result % div;
207  }
208  }

211  UInt64 next_uint64_bounded(uint64_t bound, UInt64Mask mask = true) {
212  if constexpr (is_scalar_v<T>) {
213  ENOKI_MARK_USED(mask);
214 
215  const uint64_t threshold = (~bound + (uint64_t) 1) % bound;
216 
217  while (true) {
218  uint64_t result = next_uint64();
219 
220  if (all(result >= threshold))
221  return result % bound;
222  }
223  } else {
224  const divisor_ext<uint64_t> div(bound);
225  const UInt64 threshold = (~bound + (uint64_t) 1) % div;
226 
227  UInt64 result = zero<UInt64>();
228  do {
229  result[mask] = next_uint64(mask);
230 
231  /* Keep track of which SIMD lanes have already
232  finished and stops advancing the associated PRNGs */
233  mask &= result < threshold;
234  } while (any(mask));
235 
236  return result % div;
237  }

247  void advance(const Int64 &delta_) {
248  UInt64 cur_mult = PCG32_MULT,
249  cur_plus = inc,
250  acc_mult = 1ull,
251  acc_plus = 0ull;
252 
253  /* Even though delta is an unsigned integer, we can pass a signed
254  integer to go backwards, it just goes "the long way round". */
255  UInt64 delta(delta_);
256 
257  while (delta != zero<UInt64>()) {
258  auto mask = neq(delta & UInt64(1), zero<UInt64>());
259  acc_mult = select(mask, acc_mult * cur_mult, acc_mult);
260  acc_plus = select(mask, acc_plus * cur_mult + cur_plus, acc_plus);
261  cur_plus = (cur_mult + UInt64(1)) * cur_plus;
262  cur_mult *= cur_mult;
263  delta = sr<1>(delta);
264  }
265 
266  state = acc_mult * state + acc_plus;
267  }

270  Int64 operator-(const PCG32 &other) const {
271  assert(inc == other.inc);
272 
273  UInt64 cur_mult = PCG32_MULT,
274  cur_plus = inc,
275  cur_state = other.state,
276  the_bit = 1ull,
277  distance = 0ull;
278 
279  while (state != cur_state) {
280  auto mask = neq(state & the_bit, cur_state & the_bit);
281  cur_state = select(mask, cur_state * cur_mult + cur_plus, cur_state);
282  distance = select(mask, distance | the_bit, distance);
283  assert((state & the_bit) == (cur_state & the_bit));
284  the_bit = sl<1>(the_bit);
285  cur_plus = (cur_mult + UInt64(1)) * cur_plus;
286  cur_mult *= cur_mult;
287  }
288 
289  return Int64(distance);
290  }

298  template <typename Iterator, typename T2 = T,
299  enable_if_t<is_scalar_v<T2>> = 0>
300  void shuffle(Iterator begin, Iterator end) {
301  for (Iterator it = end - 1; it > begin; --it)
302  std::iter_swap(it, begin + next_uint32_bounded((uint32_t) (it - begin + 1)));
303  }
304 
306  bool operator==(const PCG32 &other) const { return state == other.state && inc == other.inc; }
307 
309  bool operator!=(const PCG32 &other) const { return state != other.state || inc != other.inc; }

311  UInt64 state; // RNG state. All values are possible.
312  UInt64 inc; // Controls which RNG sequence (stream) is selected. Must *always* be odd.

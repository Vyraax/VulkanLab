205  return vsqrtq_f32(m);
206  #else
207  if constexpr (Approx_) {
208  const float32x4_t inf = vdupq_n_f32(std::numeric_limits<float>::infinity());
209  float32x4_t r = vrsqrteq_f32(m);
210  uint32x4_t inf_or_zero = vorrq_u32(vceqq_f32(r, inf), vceqq_f32(m, inf));
211  r = vmulq_f32(r, vrsqrtsq_f32(vmulq_f32(r, r), m));
212  r = vmulq_f32(r, vrsqrtsq_f32(vmulq_f32(r, r), m));
213  r = vmulq_f32(r, m);
214  return vbslq_f32(inf_or_zero, m, r);
215  } else {

217  }

248  template <int I0, int I1, int I2, int I3>
249  ENOKI_INLINE Derived shuffle_() const {
251  switch (I3 + I2*10 + I1*100 + I0*1000) {
252  case 0123: return m;
253  case 0000: return vdupq_lane_f32(vget_low_f32(m), 0);
254  case 1111: return vdupq_lane_f32(vget_low_f32(m), 1);
255  case 2222: return vdupq_lane_f32(vget_high_f32(m), 0);
256  case 3333: return vdupq_lane_f32(vget_high_f32(m), 1);
257  case 1032: return vrev64q_f32(m);
258  case 0101: { float32x2_t vt = vget_low_f32(m); return vcombine_f32(vt, vt); }
259  case 2323: { float32x2_t vt = vget_high_f32(m); return vcombine_f32(vt, vt); }
260  case 1010: { float32x2_t vt = vrev64_f32(vget_low_f32(m)); return vcombine_f32(vt, vt); }
261  case 3232: { float32x2_t vt = vrev64_f32(vget_high_f32(m)); return vcombine_f32(vt, vt); }
262  case 0132: return vcombine_f32(vget_low_f32(m), vrev64_f32(vget_high_f32(m)));
263  case 1023: return vcombine_f32(vrev64_f32(vget_low_f32(m)), vget_high_f32(m));
264  case 2310: return vcombine_f32(vget_high_f32(m), vrev64_f32(vget_low_f32(m)));
265  case 3201: return vcombine_f32(vrev64_f32(vget_high_f32(m)), vget_low_f32(m));
266  case 3210: return vcombine_f32(vrev64_f32(vget_high_f32(m)), vrev64_f32(vget_low_f32(m)));
267 #if defined(ENOKI_ARM_64)
268  case 0022: return vtrn1q_f32(m, m);
269  case 1133: return vtrn2q_f32(m, m);
270  case 0011: return vzip1q_f32(m, m);
271  case 2233: return vzip2q_f32(m, m);
272  case 0202: return vuzp1q_f32(m, m);
273  case 1313: return vuzp2q_f32(m, m);
274 #endif
275  case 1230: return vextq_f32(m, m, 1);
276  case 2301: return vextq_f32(m, m, 2);
277  case 3012: return vextq_f32(m, m, 3);
278 
279  default: {
280  constexpr uint64_t prec0 = detail::arm_shuffle_helper_(I0) |
281  (detail::arm_shuffle_helper_(I1) << 32);
282  constexpr uint64_t prec1 = detail::arm_shuffle_helper_(I2) |
283  (detail::arm_shuffle_helper_(I3) << 32);
284 
285  uint8x8x2_t tbl;
286  tbl.val[0] = vreinterpret_u8_f32(vget_low_f32(m));
287  tbl.val[1] = vreinterpret_u8_f32(vget_high_f32(m));
288 
289  uint8x8_t idx1 = vreinterpret_u8_u32(vcreate_u32(prec0));
290  uint8x8_t idx2 = vreinterpret_u8_u32(vcreate_u32(prec1));
291 
292  float32x2_t l = vreinterpret_f32_u8(vtbl2_u8(tbl, idx1));
293  float32x2_t h = vreinterpret_f32_u8(vtbl2_u8(tbl, idx2));
294 
295  return vcombine_f32(l, h);
296  }
297  }
298  }

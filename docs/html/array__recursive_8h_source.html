305  template <size_t... Indices1, typename T= size_t, size_t... Indices2>
306  ENOKI_INLINE Derived shuffle_impl_(std::index_sequence<Indices1...>,
307  std::index_sequence<Indices2...>) const {
308  using Int = int_array_t<Array1>;
309  Array1 a1l = a1.template shuffle_<(size_t) std::min(Size1 - 1, Indices1)...>(),
310  a1h = a2.template shuffle_<(size_t) std::max((ssize_t) 0, (ssize_t) Indices1 - (ssize_t) Size1)...>(),
311  a1f = select(Int(Indices1...) < Int(Size1), a1l, a1h);
312 
313  Array2 a2l = a1.template shuffle_<std::min(Size1 - 1, Indices2)...>(),
314  a2h = a2.template shuffle_<(size_t) std::max((ssize_t) 0, (ssize_t) Indices2 - (ssize_t) Size1)...>(),
315  a2f = select(Int(Indices2...) < Int(Size1), a2l, a2h);
316 
317  return Derived(a1f, a2f);
318  }

320  template <typename Index> ENOKI_INLINE Derived shuffle_(const Index &index) const {
321  if constexpr (Size1 != Size2) {
322  return Base::shuffle_(index);
323  } else {
324  auto il = low(index), ih = high(index);
325 
326  decltype(il) size = scalar_t<Index>(Size1);
327 
328  Array1 a1l = a1.shuffle_(il),
329  a1h = a2.shuffle_(il - size),
330  a1f = select(il < size, a1l, a1h);
331 
332  Array2 a2l = a1.shuffle_(ih),
333  a2h = a2.shuffle_(ih - size),
334  a2f = select(ih < size, a2l, a2h);
335 
336  return Derived(a1f, a2f);
337  }

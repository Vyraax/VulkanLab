22 template <typename T, enable_if_dynamic_array_t<T> = 0>
24 Array<T, 2> meshgrid(const T &x, const T &y) {
25  T X, Y;
26  set_slices(X, x.size() * y.size());
27  set_slices(Y, x.size() * y.size());
28 
29  size_t pos = 0;
30 
31  if (x.size() % T::PacketSize == 0) {
32  /* Fast path */
33 
34  for (size_t i = 0; i < y.size(); ++i) {
35  for (size_t j = 0; j < packets(x); ++j) {
36  packet(X, pos) = packet(x, j);
37  packet(Y, pos) = y.coeff(i);
38  pos++;
39  }
40  }
41  } else {
42  for (size_t i = 0; i < y.size(); ++i) {
43  for (size_t j = 0; j < x.size(); ++j) {
44  X.coeff(pos) = x.coeff(j);
45  Y.coeff(pos) = y.coeff(i);
46  pos++;
47  }
48  }
49  }
50 
51  return Array<T, 2>(std::move(X), std::move(Y));
52 }

55 template <typename Value> struct range {
56  static constexpr size_t Dimension = array_depth_v<Value> == 2 ?
57  array_size_v<Value> : 1;
58  static constexpr size_t PacketSize = array_depth_v<Value> == 2 ?
59  array_size_v<value_t<Value>> : array_size_v<Value>;
60 
61  using Scalar = scalar_t<Value>;
62  using Packet = Array<Scalar, PacketSize>;
63  using Size = Array<Scalar, Dimension>;
64 
65  struct iterator {
66  iterator(size_t index) : index(index) { }
67  iterator(size_t index, Size size)
68  : index(index), index_p(arange<Packet>()), size(size) {
69  for (size_t i = 0; i < Dimension - 1; ++i)
70  div[i] = size[i];
71  }
72 
73  bool operator==(const iterator &it) const { return it.index == index; }
74  bool operator!=(const iterator &it) const { return it.index != index; }
75 
76  iterator &operator++() {
77  index += 1;
78  index_p += Scalar(Packet::Size);
79  return *this;
80  }
81 
82  std::pair<Value, mask_t<Packet>> operator*() const {
83  if constexpr (array_depth_v<Value> == 1) {
84  return { index_p, index_p < size[0] };
85  } else {
86  Value value;
87  value[0] = index_p;
88  ENOKI_UNROLL for (size_t i = 0; i < Dimension - 1; ++i)
89  value[i + 1] = div[i](value[i]);
90  Packet offset = zero<Packet>();
91  ENOKI_UNROLL for (size_t i = Dimension - 2; ; --i) {
92  offset = size[i] * (value[i + 1] + offset);
93  value[i] -= offset;
94  if (i == 0)
95  break;
96  }
97 
98  return { value, value[Dimension - 1] < size[Dimension - 1] };
99  }
100  }
101 
102  private:
103  size_t index;
104  Packet index_p;
105  Size size;
106  divisor<Scalar> div[Dimension > 1 ? (Dimension - 1) : 1];
107  };
108 
109  template <typename... Args>
110  range(Args&&... args) : size(args...) { }
111 
112  iterator begin() {
113  return iterator(0, size);
114  }
115 
116  iterator end() {
117  return iterator((hprod(size) + Packet::Size - 1) / Packet::Size);
118  }
119 
120 private:
121  Size size;
122 };

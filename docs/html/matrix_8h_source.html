61  template <typename Value2, size_t Size2, bool Approx2, enable_if_t<Size2 == Size_> = 0>
62  ENOKI_INLINE Matrix(const Matrix<Value2, Size2, Approx2> &m)
63  : Base(m) { }
64 
66  template <size_t Size2, bool Approx2, enable_if_t<Size2 != Size_> = 0>
67  ENOKI_INLINE Matrix(const Matrix<Value_, Size2, Approx2> &m) {
68  if constexpr (Size2 > Size) {
70  for (size_t i = 0; i < Size; ++i)
71  coeff(i) = head<Size>(m.coeff(i));
72  } else {
74  using Remainder = Array<Value_, Size - Size2, Approx_>;
75  for (size_t i = 0; i < Size2; ++i)
76  coeff(i) = concat(m.coeff(i), zero<Remainder>());
77  for (size_t i = Size2; i < Size; ++i) {
78  auto col = zero<Column>();
79  col.coeff(i) = 1;
80  coeff(i) = col;
81  }
82  }
83  }
84 
85  template <typename T, enable_if_t<(array_depth_v<T> <= Base::Depth - 2)> = 0,
86  enable_if_not_matrix_t<T> = 0>
87  ENOKI_INLINE Matrix(T&& v) {
88  for (size_t i = 0; i < Size; ++i) {
89  coeff(i) = zero<Column>();
90  coeff(i, i) = v;
91  }
92  }
93 
94  template <typename T, enable_if_t<(array_depth_v<T> == Base::Depth)> = 0,
95  enable_if_not_matrix_t<T> = 0>
96  ENOKI_INLINE Matrix(T&& v) : Base(std::forward<T>(v)) { }
97 
99  template <typename... Args, enable_if_t<sizeof...(Args) == Size_ &&
100  std::conjunction_v<std::is_constructible<Column, Args>...>> = 0>
101  ENOKI_INLINE Matrix(const Args&... args) : Base(args...) { }
102 
104  template <typename... Args, enable_if_t<sizeof...(Args) == Size_ * Size_ &&
105  std::conjunction_v<std::is_constructible<Entry, Args>...>> = 0>
106  ENOKI_INLINE Matrix(const Args&... args) {
107  alignas(alignof(Column)) Entry values[sizeof...(Args)] = { Entry(args)... };
108  for (size_t j = 0; j < Size; ++j)
109  for (size_t i = 0; i < Size; ++i)
110  coeff(j, i) = values[i * Size + j];
111  }

116  }

123  ENOKI_INLINE Column& col(size_t index) { return coeff(index); }
124  ENOKI_INLINE const Column& col(size_t index) const { return coeff(index); }
125 
126  ENOKI_INLINE Column row(size_t index) const {
127  using Index = Array<uint32_t, Size>;
128  return gather<Column>(coeff(0).data() + index,
129  arange<Index>() * uint32_t(Size));

131 
133  ENOKI_INLINE decltype(auto) operator()(size_t i, size_t j) { return coeff(j, i); }
134 
136  ENOKI_INLINE decltype(auto) operator()(size_t i, size_t j) const { return coeff(j, i); }

138  static ENOKI_INLINE Derived zero_(size_t size) {
139  Derived result;
140  for (size_t i = 0; i < Size; ++i)
141  result.coeff(i) = zero<Column>(size);
142  return result;
143  }

145  static ENOKI_INLINE Derived empty_(size_t size) {
146  Derived result;
147  for (size_t i = 0; i < Size; ++i)
148  result.coeff(i) = empty<Column>(size);

152  template <typename T> ENOKI_INLINE static Matrix full_(const T &value, size_t size) {
153  return Array<Column, Size>::full_(value, size);
154  }

157 template <typename T0, typename T1, size_t Size, bool Approx0, bool Approx1,
158  typename Result = Matrix<expr_t<T0, T1>, Size, Approx0 && Approx1>,
159  typename Column = column_t<Result>>
160 ENOKI_INLINE Result operator*(const Matrix<T0, Size, Approx0> &m0,
161  const Matrix<T1, Size, Approx1> &m1) {
162  Result result;
163  /* 4x4 case reduced to 4 multiplications, 12 fused multiply-adds,
164  and 16 broadcasts (also fused on AVX512VL) */
165  for (size_t j = 0; j < Size; ++j) {
166  Column sum = m0.coeff(0) * Column::full_(m1(0, j), 1);
167  for (size_t i = 1; i < Size; ++i)
168  sum = fmadd(m0.coeff(i), Column::full_(m1(i, j), 1), sum);
169  result.coeff(j) = sum;
170  }
171 
172  return result;
173 }

175 template <typename T0, typename T1, size_t Size, bool Approx,
176  typename Value = expr_t<T0, T1>>
177 ENOKI_INLINE auto operator*(const Matrix<T0, Size, Approx> &m, const T1 &s) {
178  if constexpr (array_size_v<T1> == Size && !std::is_same_v<T1, T0>) {
179  using Return = column_t<Matrix<expr_t<T0, value_t<T1>>, Size>>;
180  Return sum = m.coeff(0) * Return::full_(s.coeff(0), 1);
181  for (size_t i = 1; i < Size; ++i)
182  sum = fmadd(m.coeff(i), Return::full_(s.coeff(i), 1), sum);
183  return sum;
184  } else {
185  using Result = Matrix<expr_t<T0, T1>, Size, Approx>;
186  return Result(Array<Array<expr_t<T0>, Size>, Size>(m) *
187  full<Array<Array<scalar_t<T1>, Size>, Size>>(s));
188  }
189 }

214 ENOKI_INLINE expr_t<Value> frob(const Matrix<Value, Size, Approx> &matrix) {
215  expr_t<column_t<Matrix<Value, Size, Approx>>> result = sqr(matrix.coeff(0));

287 ENOKI_INLINE Matrix<E, 3, Approx>
288 inverse_transpose(const Matrix<T, 3, Approx> &m) {
289  using Vector = Array<E, 3>;
290 
291  Vector col0 = m.coeff(0),
292  col1 = m.coeff(1),
293  col2 = m.coeff(2);
294 
295  Vector row0 = cross(col1, col2);
296  Vector row1 = cross(col2, col0);
297  Vector row2 = cross(col0, col1);
298 
299  Vector inv_det = Vector(rcp<Vector::Approx>(dot(col0, row0)));
300 
301  return Matrix<E, 3, Approx>(
302  row0 * inv_det,
303  row1 * inv_det,
304  row2 * inv_det
305  );
306 }

318 template <typename T, bool Approx, typename E = expr_t<T>>
319 ENOKI_INLINE Matrix<E, 4, Approx>
320 inverse_transpose(const Matrix<T, 4, Approx> &m) {
321  using Vector = Array<E, 4>;
322 
323  Vector col0 = m.coeff(0), col1 = m.coeff(1),
324  col2 = m.coeff(2), col3 = m.coeff(3);
325 
326  col1 = shuffle<2, 3, 0, 1>(col1);
327  col3 = shuffle<2, 3, 0, 1>(col3);
328 
329  Vector tmp, row0, row1, row2, row3;
330 
331  tmp = shuffle<1, 0, 3, 2>(col2 * col3);
332  row0 = col1 * tmp;
333  row1 = col0 * tmp;
334  tmp = shuffle<2, 3, 0, 1>(tmp);
335  row0 = fmsub(col1, tmp, row0);
336  row1 = shuffle<2, 3, 0, 1>(fmsub(col0, tmp, row1));
337 
338  tmp = shuffle<1, 0, 3, 2>(col1 * col2);
339  row0 = fmadd(col3, tmp, row0);
340  row3 = col0 * tmp;
341  tmp = shuffle<2, 3, 0, 1>(tmp);
342  row0 = fnmadd(col3, tmp, row0);
343  row3 = shuffle<2, 3, 0, 1>(fmsub(col0, tmp, row3));
344 
345  tmp = shuffle<1, 0, 3, 2>(shuffle<2, 3, 0, 1>(col1) * col3);
346  col2 = shuffle<2, 3, 0, 1>(col2);
347  row0 = fmadd(col2, tmp, row0);
348  row2 = col0 * tmp;
349  tmp = shuffle<2, 3, 0, 1>(tmp);
350  row0 = fnmadd(col2, tmp, row0);
351  row2 = shuffle<2, 3, 0, 1>(fmsub(col0, tmp, row2));
352 
353  tmp = shuffle<1, 0, 3, 2>(col0 * col1);
354  row2 = fmadd(col3, tmp, row2);
355  row3 = fmsub(col2, tmp, row3);
356  tmp = shuffle<2, 3, 0, 1>(tmp);
357  row2 = fmsub(col3, tmp, row2);
358  row3 = fnmadd(col2, tmp, row3);
359 
360  tmp = shuffle<1, 0, 3, 2>(col0 * col3);
361  row1 = fnmadd(col2, tmp, row1);
362  row2 = fmadd(col1, tmp, row2);
363  tmp = shuffle<2, 3, 0, 1>(tmp);
364  row1 = fmadd(col2, tmp, row1);
365  row2 = fnmadd(col1, tmp, row2);
366 
367  tmp = shuffle<1, 0, 3, 2>(col0 * col2);
368  row1 = fmadd(col3, tmp, row1);
369  row3 = fnmadd(col1, tmp, row3);
370  tmp = shuffle<2, 3, 0, 1>(tmp);
371  row1 = fnmadd(col3, tmp, row1);
372  row3 = fmadd(col1, tmp, row3);
373 
374  Vector inv_det = Vector(rcp<Vector::Approx>(dot(col0, row0)));
375 
376  return Matrix<E, 4, Approx>(
377  row0 * inv_det, row1 * inv_det,
378  row2 * inv_det, row3 * inv_det
379  );
380 }

387 template <typename T, bool Approx, typename E = expr_t<T>>
388 ENOKI_INLINE E det(const Matrix<T, 4, Approx> &m) {
389  using Vector = Array<E, 4>;
390 
391  Vector col0 = m.coeff(0), col1 = m.coeff(1),
392  col2 = m.coeff(2), col3 = m.coeff(3);
393 
394  col1 = shuffle<2, 3, 0, 1>(col1);
395  col3 = shuffle<2, 3, 0, 1>(col3);
396 
397  Vector tmp, row0;
398 
399  tmp = shuffle<1, 0, 3, 2>(col2 * col3);
400  row0 = col1 * tmp;
401  tmp = shuffle<2, 3, 0, 1>(tmp);
402  row0 = fmsub(col1, tmp, row0);
403 
404  tmp = shuffle<1, 0, 3, 2>(col1 * col2);
405  row0 = fmadd(col3, tmp, row0);
406  tmp = shuffle<2, 3, 0, 1>(tmp);
407  row0 = fnmadd(col3, tmp, row0);
408 
409  col1 = shuffle<2, 3, 0, 1>(col1);
410  col2 = shuffle<2, 3, 0, 1>(col2);
411  tmp = shuffle<1, 0, 3, 2>(col1 * col3);
412  row0 = fmadd(col2, tmp, row0);
413  tmp = shuffle<2, 3, 0, 1>(tmp);
414  row0 = fnmadd(col2, tmp, row0);
415 
416  return dot(col0, row0);
417 }

419 template <typename Value, size_t Size, bool Approx, RoundingMode Mode, bool IsMask_, typename Derived>
420 ENOKI_INLINE auto transpose(const StaticArrayBase<Value, Size, Approx, Mode, IsMask_, Derived> &a) {
421  static_assert(Value::Size == Size && array_depth<Derived>::value >= 2,
422  "Array must be a square matrix!");
423  using Column = value_t<Derived>;
424 
425  if constexpr (Column::IsNative) {
426  #if defined(ENOKI_X86_SSE42)
427  if constexpr (std::is_same_v<value_t<Column>, float> && Size == 3) {
428  __m128 c0 = a.derived().coeff(0).m,
429  c1 = a.derived().coeff(1).m,
430  c2 = a.derived().coeff(2).m;
431 
432  __m128 t0 = _mm_unpacklo_ps(c0, c1);
433  __m128 t1 = _mm_unpacklo_ps(c2, c2);
434  __m128 t2 = _mm_unpackhi_ps(c0, c1);
435  __m128 t3 = _mm_unpackhi_ps(c2, c2);
436 
437  return Derived(
438  _mm_movelh_ps(t0, t1),
439  _mm_movehl_ps(t1, t0),
440  _mm_movelh_ps(t2, t3)
441  );
442  } else if constexpr (std::is_same_v<value_t<Column>, float> && Size == 4) {
443  __m128 c0 = a.derived().coeff(0).m, c1 = a.derived().coeff(1).m,
444  c2 = a.derived().coeff(2).m, c3 = a.derived().coeff(3).m;
445 
446  __m128 t0 = _mm_unpacklo_ps(c0, c1);
447  __m128 t1 = _mm_unpacklo_ps(c2, c3);
448  __m128 t2 = _mm_unpackhi_ps(c0, c1);
449  __m128 t3 = _mm_unpackhi_ps(c2, c3);
450 
451  return Derived(
452  _mm_movelh_ps(t0, t1),
453  _mm_movehl_ps(t1, t0),
454  _mm_movelh_ps(t2, t3),
455  _mm_movehl_ps(t3, t2)
456  );
457  }
458  #endif
459 
460  #if defined(ENOKI_X86_AVX)
461  if constexpr (std::is_same_v<value_t<Column>, double> && Size == 3) {
462  __m256d c0 = a.derived().coeff(0).m,
463  c1 = a.derived().coeff(1).m,
464  c2 = a.derived().coeff(2).m;
465 
466  __m256d t3 = _mm256_shuffle_pd(c2, c2, 0b0000),
467  t2 = _mm256_shuffle_pd(c2, c2, 0b1111),
468  t1 = _mm256_shuffle_pd(c0, c1, 0b0000),
469  t0 = _mm256_shuffle_pd(c0, c1, 0b1111);
470 
471  return Derived(
472  _mm256_permute2f128_pd(t1, t3, 0b0010'0000),
473  _mm256_permute2f128_pd(t0, t2, 0b0010'0000),
474  _mm256_permute2f128_pd(t1, t3, 0b0011'0001)
475  );
476  } else if constexpr (std::is_same_v<value_t<Column>, double> && Size == 4) {
477  __m256d c0 = a.derived().coeff(0).m, c1 = a.derived().coeff(1).m,
478  c2 = a.derived().coeff(2).m, c3 = a.derived().coeff(3).m;
479 
480  __m256d t3 = _mm256_shuffle_pd(c2, c3, 0b0000),
481  t2 = _mm256_shuffle_pd(c2, c3, 0b1111),
482  t1 = _mm256_shuffle_pd(c0, c1, 0b0000),
483  t0 = _mm256_shuffle_pd(c0, c1, 0b1111);
484 
485  return Derived(
486  _mm256_permute2f128_pd(t1, t3, 0b0010'0000),
487  _mm256_permute2f128_pd(t0, t2, 0b0010'0000),
488  _mm256_permute2f128_pd(t1, t3, 0b0011'0001),
489  _mm256_permute2f128_pd(t0, t2, 0b0011'0001)
490  );
491  }
492  #endif
493 
494  #if defined(ENOKI_ARM_NEON)
495  if constexpr (std::is_same_v<value_t<Column>, float> && Size == 3) {
496  float32x4x2_t v01 = vtrnq_f32(a.derived().coeff(0).m, a.derived().coeff(1).m);
497  float32x4x2_t v23 = vtrnq_f32(a.derived().coeff(2).m, a.derived().coeff(2).m);
498 
499  return Derived(
500  vcombine_f32(vget_low_f32 (v01.val[0]), vget_low_f32 (v23.val[0])),
501  vcombine_f32(vget_low_f32 (v01.val[1]), vget_low_f32 (v23.val[1])),
502  vcombine_f32(vget_high_f32(v01.val[0]), vget_high_f32(v23.val[0]))
503  );
504  } else if constexpr (std::is_same_v<value_t<Column>, float> && Size == 4) {
505  float32x4x2_t v01 = vtrnq_f32(a.derived().coeff(0).m, a.derived().coeff(1).m);
506  float32x4x2_t v23 = vtrnq_f32(a.derived().coeff(2).m, a.derived().coeff(3).m);
507 
508  return Derived(

529 std::pair<Matrix, Matrix> ENOKI_INLINE
530 polar_decomp(const enoki::Matrix<T, Size, Approx> &A, size_t it = 10) {
531  using Arr = Array<Array<Expr, Size>, Size>;
532  Matrix Q = A;
533  for (size_t i = 0; i < it; ++i) {
534  Matrix Qi = inverse_transpose(Q);
535  Expr gamma = sqrt(frob(Qi) / frob(Q));
536  Q = fmadd(Arr(Q), gamma * .5f, Arr(Qi) * (rcp(gamma) * 0.5f));
537  }
538  return std::make_pair(Q, transpose(Q) * A);
539 }

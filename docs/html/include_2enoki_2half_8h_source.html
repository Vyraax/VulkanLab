77  /*
78  Value float32<->float16 conversion code by Paul A. Tessier (@Phernost)
79  Used with permission by the author, who released this code into the public domain
80  */
81  union Bits {
82  float f;
83  int32_t si;
84  uint32_t ui;
85  };
86 
87  static constexpr int const shift = 13;
88  static constexpr int const shiftSign = 16;
89 
90  static constexpr int32_t const infN = 0x7F800000; // flt32 infinity
91  static constexpr int32_t const maxN = 0x477FE000; // max flt16 normal as a flt32
92  static constexpr int32_t const minN = 0x38800000; // min flt16 normal as a flt32
93  static constexpr int32_t const signN = (int32_t) 0x80000000; // flt32 sign bit
94 
95  static constexpr int32_t const infC = infN >> shift;
96  static constexpr int32_t const nanN = (infC + 1) << shift; // minimum flt16 nan as a flt32
97  static constexpr int32_t const maxC = maxN >> shift;
98  static constexpr int32_t const minC = minN >> shift;
99  static constexpr int32_t const signC = signN >> shiftSign; // flt16 sign bit
100 
101  static constexpr int32_t const mulN = 0x52000000; // (1 << 23) / minN
102  static constexpr int32_t const mulC = 0x33800000; // minN / (1 << (23 - shift))
103 
104  static constexpr int32_t const subC = 0x003FF; // max flt32 subnormal down shifted
105  static constexpr int32_t const norC = 0x00400; // min flt32 normal down shifted
106 
107  static constexpr int32_t const maxD = infC - maxC - 1;
108  static constexpr int32_t const minD = minC - subC - 1;
109 
110 public:
111  static uint16_t float32_to_float16(float value) {
112  #if defined(ENOKI_X86_F16C)
113  return (uint16_t) _mm_cvtsi128_si32(
114  _mm_cvtps_ph(_mm_set_ss(value), _MM_FROUND_CUR_DIRECTION));
115  #elif defined(ENOKI_ARM_NEON)
116  return memcpy_cast<uint16_t>((__fp16) value);
117  #else
118  Bits v, s;
119  v.f = value;
120  uint32_t sign = (uint32_t) (v.si & signN);
121  v.si ^= sign;
122  sign >>= shiftSign; // logical shift
123  s.si = mulN;
124  s.si = (int32_t) (s.f * v.f); // correct subnormals
125  v.si ^= (s.si ^ v.si) & -(minN > v.si);
126  v.si ^= (infN ^ v.si) & -((infN > v.si) & (v.si > maxN));
127  v.si ^= (nanN ^ v.si) & -((nanN > v.si) & (v.si > infN));
128  v.ui >>= shift; // logical shift
129  v.si ^= ((v.si - maxD) ^ v.si) & -(v.si > maxC);
130  v.si ^= ((v.si - minD) ^ v.si) & -(v.si > subC);
131  return (uint16_t) (v.ui | sign);
132  #endif
133  }
134 
135  static float float16_to_float32(uint16_t value) {
136  #if defined(ENOKI_X86_F16C)
137  return _mm_cvtss_f32(_mm_cvtph_ps(_mm_cvtsi32_si128((int32_t) value)));
138  #elif defined(ENOKI_ARM_NEON)
139  return (float) memcpy_cast<__fp16>(value);
140  #else
141  Bits v;
142  v.ui = value;
143  int32_t sign = v.si & signC;
144  v.si ^= sign;
145  sign <<= shiftSign;
146  v.si ^= ((v.si + minD) ^ v.si) & -(v.si > subC);
147  v.si ^= ((v.si + maxD) ^ v.si) & -(v.si > maxC);
148  Bits s;
149  s.si = mulC;
150  s.f *= float(v.si);
151  int32_t mask = -(norC > v.si);
152  v.si <<= shift;
153  v.si ^= (s.si ^ v.si) & mask;
154  v.si |= sign;
155  return v.f;
156  #endif
157  }
158 };

48 // Python loads modules by default with dlopen with the RTLD_LOCAL flag; under libc++ and possibly
49 // other STLs, this means `typeid(A)` from one module won't equal `typeid(A)` from another module
50 // even when `A` is the same, non-hidden-visibility type (e.g. from a common include). Under
51 // libstdc++, this doesn't happen: equality and the type_index hash are based on the type name,
52 // which works. If not under a known-good stl, provide our own name-based hash and equality
53 // functions that use the type name.
54 #if defined(__GLIBCXX__)
55 inline bool same_type(const std::type_info &lhs, const std::type_info &rhs) { return lhs == rhs; }
56 using type_hash = std::hash<std::type_index>;
57 using type_equal_to = std::equal_to<std::type_index>;
58 #else
59 inline bool same_type(const std::type_info &lhs, const std::type_info &rhs) {
60  return lhs.name() == rhs.name() || std::strcmp(lhs.name(), rhs.name()) == 0;
61 }

63 struct type_hash {
64  size_t operator()(const std::type_index &t) const {
65  size_t hash = 5381;
66  const char *ptr = t.name();
67  while (auto c = static_cast<unsigned char>(*ptr++))
68  hash = (hash * 33) ^ c;
69  return hash;
70  }
71 };

75  return lhs.name() == rhs.name() || std::strcmp(lhs.name(), rhs.name()) == 0;
76  }

83 struct overload_hash {
84  inline size_t operator()(const std::pair<const PyObject *, const char *>& v) const {
85  size_t value = std::hash<const void *>()(v.first);
86  value ^= std::hash<const void *>()(v.second) + 0x9e3779b9 + (value<<6) + (value>>2);
87  return value;
88  }

203 inline void translate_exception(std::exception_ptr p) {
204  try {
205  if (p) std::rethrow_exception(p);
206  } catch (error_already_set &e) { e.restore(); return;
207  } catch (const builtin_exception &e) { e.set_error(); return;
208  } catch (const std::bad_alloc &e) { PyErr_SetString(PyExc_MemoryError, e.what()); return;
209  } catch (const std::domain_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return;
210  } catch (const std::invalid_argument &e) { PyErr_SetString(PyExc_ValueError, e.what()); return;
211  } catch (const std::length_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return;
212  } catch (const std::out_of_range &e) { PyErr_SetString(PyExc_IndexError, e.what()); return;
213  } catch (const std::range_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return;
214  } catch (const std::overflow_error &e) { PyErr_SetString(PyExc_OverflowError, e.what()); return;
215  } catch (const std::exception &e) { PyErr_SetString(PyExc_RuntimeError, e.what()); return;
216  } catch (...) {
217  PyErr_SetString(PyExc_RuntimeError, "Caught an unknown exception!");
218  return;
219  }
220 }

233 PYBIND11_NOINLINE inline internals &get_internals() {
234  auto **&internals_pp = get_internals_pp();
235  if (internals_pp && *internals_pp)
236  return **internals_pp;

238  // Ensure that the GIL is held since we will need to make Python calls.
239  // Cannot use py::gil_scoped_acquire here since that constructor calls get_internals.
240  struct gil_scoped_acquire_local {
241  gil_scoped_acquire_local() : state (PyGILState_Ensure()) {}
242  ~gil_scoped_acquire_local() { PyGILState_Release(state); }
243  const PyGILState_STATE state;
244  } gil;
245 
246  constexpr auto *id = PYBIND11_INTERNALS_ID;
247  auto builtins = handle(PyEval_GetBuiltins());
248  if (builtins.contains(id) && isinstance<capsule>(builtins[id])) {
249  internals_pp = static_cast<internals **>(capsule(builtins[id]));
250 
251  // We loaded builtins through python's builtins, which means that our `error_already_set`
252  // and `builtin_exception` may be different local classes than the ones set up in the
253  // initial exception translator, below, so add another for our local exception classes.
254  //
255  // libstdc++ doesn't require this (types there are identified only by name)
256 #if !defined(__GLIBCXX__)
257  (*internals_pp)->registered_exception_translators.push_front(&translate_local_exception);
258 #endif
259  } else {
260  if (!internals_pp) internals_pp = new internals*();
261  auto *&internals_ptr = *internals_pp;
262  internals_ptr = new internals();
263 #if defined(WITH_THREAD)
264  PyEval_InitThreads();
265  PyThreadState *tstate = PyThreadState_Get();
266  #if PY_VERSION_HEX >= 0x03070000
267  internals_ptr->tstate = PyThread_tss_alloc();
268  if (!internals_ptr->tstate || PyThread_tss_create(internals_ptr->tstate))
269  pybind11_fail("get_internals: could not successfully initialize the TSS key!");
270  PyThread_tss_set(internals_ptr->tstate, tstate);
271  #else
272  internals_ptr->tstate = PyThread_create_key();
273  if (internals_ptr->tstate == -1)
274  pybind11_fail("get_internals: could not successfully initialize the TLS key!");
275  PyThread_set_key_value(internals_ptr->tstate, tstate);
276  #endif
277  internals_ptr->istate = tstate->interp;
278 #endif
279  builtins[id] = capsule(internals_pp);
280  internals_ptr->registered_exception_translators.push_front(&translate_exception);
281  internals_ptr->static_property_type = make_static_property_type();
282  internals_ptr->default_metaclass = make_default_metaclass();
283  internals_ptr->instance_base = make_object_base_type(internals_ptr->default_metaclass);
284  }
285  return **internals_pp;
286 }

325 template<typename T>
326 T &get_or_create_shared_data(const std::string &name) {
327  auto &internals = detail::get_internals();
328  auto it = internals.shared_data.find(name);
329  T *ptr = (T *) (it != internals.shared_data.end() ? it->second : nullptr);

331  ptr = new T();
332  internals.shared_data[name] = ptr;

334  return *ptr;

47 template <typename Value>
48 struct type_caster<Value, std::enable_if_t<enoki::is_array_v<Value> &&
49  !enoki::is_cuda_array_v<Value>>> {
50  using Scalar = std::conditional_t<Value::IsMask, bool, enoki::scalar_t<Value>>;
51  static constexpr bool IsComplex = Value::IsComplex;
52 
53  bool load(handle src, bool convert) {
54  if (src.is_none()) {
55  is_none = true;
56  return true;
57  }
58 
59  if (!isinstance<array_t<Scalar>>(src)) {
60  if (!convert)
61  return false;
62 
64  if (strncmp(((PyTypeObject *) src.get_type().ptr())->tp_name, "enoki.", 6) == 0)
65  return false;
66  }
67 
68  constexpr size_t ndim = enoki::array_depth_v<Value>;
69 
70  array arr = reinterpret_borrow<array>(src);
71  if constexpr (IsComplex) {
72  auto np = module::import("numpy");
73  try {
74  arr = np.attr("asarray")(arr, sizeof(Scalar) == 4 ? "c8" : "c16", "F");
75  arr = np.attr("expand_dims")(arr, -1).attr("view")(
76  sizeof(Scalar) == 4 ? "f4" : "f8");
77  } catch (const error_already_set &) {
78  return false;
79  }
80  }
81 
82  arr = array_t<Scalar, array::f_style | array::forcecast>::ensure(arr);
83  if (!arr)
84  return false;
85 
86  if (ndim != arr.ndim() && !((arr.ndim() == 0 || (arr.ndim() == 1 && IsComplex)) && convert))
87  return false;
88 
89  std::array<size_t, ndim> shape;
90  std::fill(shape.begin(), shape.end(), (size_t) 1);
91  std::reverse_copy(arr.shape(), arr.shape() + arr.ndim(), shape.begin());
92 
93  try {
94  enoki::set_shape(value, shape);
95  } catch (const std::length_error &) {
96  return false;
97  }
98 
99  const Scalar *buf = static_cast<const Scalar *>(arr.data());
100  read_buffer(buf, value);
101 
102  return true;
103  }
104 
105  static handle cast(const Value *src, return_value_policy policy, handle parent) {
106  if (!src)
107  return pybind11::none();
108  return cast(*src, policy, parent);
109  }

111  static handle cast(const Value &src, return_value_policy /* policy */, handle /* parent */) {
112  if (enoki::ragged(src))
113  throw type_error("Ragged arrays are not supported!");
114 
115  auto shape = enoki::shape(src);
116  std::reverse(shape.begin(), shape.end());
117  decltype(shape) stride;
118 
119  stride[0] = sizeof(Scalar);
120  for (size_t i = 1; i < shape.size(); ++i)
121  stride[i] = shape[i - 1] * stride[i - 1];
122 
123  array arr(pybind11::dtype::of<Scalar>(),
124  std::vector<ssize_t>(shape.begin(), shape.end()),
125  std::vector<ssize_t>(stride.begin(), stride.end()));
126 
127  Scalar *buf = static_cast<Scalar *>(arr.mutable_data());
128  write_buffer(buf, src);
129 
130  if constexpr (IsComplex) {
131  auto np = module::import("numpy");
132  arr = np.attr("ascontiguousarray")(arr).attr("view")(
133  sizeof(Scalar) == 4 ? "c8" : "c16").attr("squeeze")(-1);
134  }
135 
136  return arr.release();
137  }

163 private:
164  template <typename T> static ENOKI_INLINE void write_buffer(Scalar *&buf, const T &value) {
165  if constexpr (!enoki::is_array_v<enoki::value_t<T>>) {
166  if constexpr (!enoki::is_mask_v<T>) {
167  memcpy(buf, value.data(), sizeof(enoki::value_t<T>) * value.size());
168  buf += value.size();
169  } else {
170  for (size_t i = 0, size = value.size(); i < size; ++i)
171  *buf++ = value.coeff(i);
172  }
173  } else {
174  for (size_t i = 0, size = value.size(); i < size; ++i)
175  write_buffer(buf, value.coeff(i));
176  }
177  }
178 
179  template <typename T>
180  static ENOKI_INLINE void read_buffer(const Scalar *&buf, T &value) {
181  if constexpr (!enoki::is_array_v<enoki::value_t<T>>) {
182  if constexpr (!enoki::is_mask_v<T>) {
183  memcpy(value.data(), buf, sizeof(enoki::value_t<T>) * value.size());
184  buf += value.size();
185  } else {
186  if constexpr (!enoki::is_dynamic_array_v<T>) {
187  enoki::Array<bool, T::Size> value2;
188  for (size_t i = 0, size = value2.size(); i < size; ++i)
189  value2.coeff(i) = *buf++;
190  value = enoki::reinterpret_array<T>(value2);
191  } else {
192  const Scalar *end = buf + value.size();
193  for (size_t i = 0; i < enoki::packets(value); ++i) {
194  enoki::Array<bool, T::Packet::Size> value2;
195  for (size_t j = 0; j < T::Packet::Size && buf != end; ++j)
196  value2.coeff(j) = *buf++;
197  enoki::packet(value, i) = enoki::reinterpret_array<typename T::Packet>(value2);
198  }
199  }
200  }
201  } else {
202  for (size_t i = 0, size = value.size(); i < size; ++i)
203  read_buffer(buf, value.coeff(i));
204  }
205  }

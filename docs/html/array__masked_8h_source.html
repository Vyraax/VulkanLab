91 template <typename T, typename Mask>
92 ENOKI_INLINE auto masked(T &value, const Mask &mask) {
93  if constexpr (std::is_same_v<Mask, bool>)
94  return detail::MaskedValue<T>{ value, mask };
95  else
96  return struct_support_t<T>::masked(value, mask);

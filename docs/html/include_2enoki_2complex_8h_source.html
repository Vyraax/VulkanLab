78 template <typename T, bool Approx> ENOKI_INLINE Complex<expr_t<T>, Approx> rcp(const Complex<T, Approx> &z) {
79  auto scale = rcp<Approx>(squared_norm(z));
80  return Complex<expr_t<T>, Approx>(
81  real(z) * scale,
82  -imag(z) * scale
83  );

86 template <typename T0, typename T1, bool Approx0, bool Approx1,
87  typename Value = expr_t<T0, T1>, typename Result = Complex<Value, Approx0 && Approx1>>
88 ENOKI_INLINE Result operator*(const Complex<T0, Approx0> &z0, const Complex<T1, Approx1> &z1) {
89  using Base = Array<Value, 2, Approx0 && Approx1>;
90  Base z1_perm = shuffle<1, 0>(z1);
91  Base z0_im = shuffle<1, 1>(z0);
92  Base z0_re = shuffle<0, 0>(z0);
93  return fmaddsub(z0_re, z1, z0_im * z1_perm);
94 }

145 sincos_arg_diff(const Complex<T1> &z1, const Complex<T2> &z2) {
146  Expr normalization = rsqrt(squared_norm(z1) * squared_norm(z2));
147  Complex<Expr> value = z1 * conj(z2) * normalization;

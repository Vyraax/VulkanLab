197  virtual bool mouse_button_event(const Vector2i &p, int button, bool down,
198  int modifiers) override {
199  if ((m_editable || m_spinnable) && down)
200  m_mouse_down_value = value();
201 
202  SpinArea area = spin_area(p);
203  if (m_spinnable && area != SpinArea::None && down && !focused()) {
204  if (area == SpinArea::Top) {
205  set_value(value() + m_value_increment);
206  if (m_callback)
207  m_callback(m_value);
208  } else if (area == SpinArea::Bottom) {
209  set_value(value() - m_value_increment);
210  if (m_callback)
211  m_callback(m_value);
212  }
213  return true;
214  }
215 
216  return TextBox::mouse_button_event(p, button, down, modifiers);
217  }

219  virtual bool mouse_drag_event(const Vector2i &p, const Vector2i &rel, int button,
220  int modifiers) override {
221  if (TextBox::mouse_drag_event(p, rel, button, modifiers))
222  return true;
223 
224  if (m_spinnable && !focused() && button == 2 /* 1 << GLFW_MOUSE_BUTTON_2 */ &&
225  m_mouse_down_pos.x() != -1) {
226  int value_delta = static_cast<int>((p.x() - m_mouse_down_pos.x()) / float(10));
227  set_value(m_mouse_down_value + value_delta * m_value_increment);
228  if (m_callback)
229  m_callback(m_value);
230  return true;
231  }
232  return false;
233  }

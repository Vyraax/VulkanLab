35 enum NVGcreateFlags {
36  // Flag indicating if geometry based anti-aliasing is used (may not be
37  // needed when using MSAA).
38  NVG_ANTIALIAS = 1 << 0,
39  // Flag indicating if strokes should be drawn using stencil buffer.
40  // The rendering will be a little slower, but path overlaps
41  // (i.e. self-intersecting or sharp turns) will be drawn just once.
42  NVG_STENCIL_STROKES = 1 << 1,
43  // Flag indicating if double buffering scheme is used.
44  NVG_DOUBLE_BUFFER = 1 << 12,
45  // Flag indicating if triple buffering scheme is used.
46  NVG_TRIPLE_BUFFER = 1 << 13,
47  // Flag indicating that additional debug checks are done.
48  NVG_DEBUG = 1 << 2,
49 };

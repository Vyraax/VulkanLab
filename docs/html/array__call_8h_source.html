118 template <typename Storage_> struct call_support_base {
119  using Storage = Storage_;
120  using InstancePtr = value_t<Storage_>;
121  using Mask = mask_t<Storage_>;
122  call_support_base(const Storage &self) : self(self) { }
123  const Storage &self;
124 
125  template <typename Func, typename InputMask,
126  typename Tuple, size_t ... Indices>
127  ENOKI_INLINE auto dispatch(Func func, InputMask mask_, Tuple tuple,
128  std::index_sequence<Indices...>) const {
129  Mask mask = Mask(mask_) & neq(self, nullptr);
130 
131  using FuncResult = decltype(func(
132  std::declval<InstancePtr>(),
133  mask,
134  std::get<Indices>(tuple)...
135  ));
136 
137  if constexpr (!std::is_void_v<FuncResult>) {
138  using Result = typename vectorize_result<Mask, FuncResult>::type;
139  Result result = zero<Result>(self.size());
140 
141  if constexpr (!is_cuda_array_v<Storage>) {
142  while (any(mask)) {
143  InstancePtr value = extract(self, mask);
144  Mask active = mask & eq(self, value);
145  mask = andnot(mask, active);
146  masked(result, active) = func(value, active, std::get<Indices>(tuple)...);
147  }
148  } else {
149  auto partitioned = partition(self & mask);
150 
151  if (partitioned.size() == 1 && partitioned[0].first != nullptr) {
152  result = func(partitioned[0].first, true,
153  std::get<Indices>(tuple)...);
154  } else {
155  for (auto [value, permutation] : partitioned) {
156  if (value == nullptr)
157  continue;
158 
159  Result temp = func(value, true, gather_helper(
160  std::get<Indices>(tuple), permutation)...);
161 
162  scatter<0, true, true>(result, temp, permutation);
163  }
164  }
165  }
166 
167  return result;
168  } else {
169  if constexpr (!is_cuda_array_v<Storage>) {
170  while (any(mask)) {
171  InstancePtr value = extract(self, mask);
172  Mask active = mask & eq(self, value);
173  mask = andnot(mask, active);
174  func(value, active, std::get<Indices>(tuple)...);
175  }
176  } else {
177  auto partitioned = partition(self & mask);
178 
179  if (partitioned.size() == 1 && partitioned[0].first != nullptr) {
180  func(partitioned[0].first, true, std::get<Indices>(tuple)...);
181  } else {
182  for (auto [value, permutation] : partitioned) {
183  if (value == nullptr)
184  continue;
185 
186  func(value, true, gather_helper(
187  std::get<Indices>(tuple), permutation)...);
188  }
189  }
190  }
191  }
192  }
193 };

237 #define ENOKI_CALL_SUPPORT_METHOD(func) \
238 private: \
239  template <typename... Args> \
240  using __##func##_t = \
241  decltype(std::declval<InstancePtr>()->func(std::declval<Args>()...)); \
242  \
243 public: \
244  template <typename... Args> auto func(Args&&... args) const { \
245  auto lambda = [](InstancePtr instance, const Mask &mask, \
246  auto &&... a) ENOKI_INLINE_LAMBDA { \
247  ENOKI_MARK_USED(mask); \
248  /* Does the method accept a mask argument? If so, provide. */ \
249  if constexpr (detail::is_callable_v<__##func##_t, decltype(a)..., \
250  Mask>) \
251  return instance->func(a..., mask); \
252  else \
253  return instance->func(a...); \
254  }; \
255  /* Was a mask provided to this function? If not, set to all ones. */ \
256  auto args_tuple = std::tie(args...); \
257  if constexpr (detail::last_of(is_mask_v<Args>...)) { \
258  return Base::dispatch( \
259  lambda, std::get<sizeof...(Args) - 1>(args_tuple), args_tuple, \
260  std::make_index_sequence<sizeof...(Args) - 1>()); \
261  } else { \
262  return Base::dispatch( \
263  lambda, true, args_tuple, \
264  std::make_index_sequence<sizeof...(Args)>()); \
265  } \
266  }

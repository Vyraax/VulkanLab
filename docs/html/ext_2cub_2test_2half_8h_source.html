74  __host__ __device__ __forceinline__
75  half_t(float a)
76  {
77  // Stolen from Norbert Juffa
78  uint32_t ia = *reinterpret_cast<uint32_t*>(&a);
79  uint16_t ir;
80 
81  ir = (ia >> 16) & 0x8000;
82 
83  if ((ia & 0x7f800000) == 0x7f800000)
84  {
85  if ((ia & 0x7fffffff) == 0x7f800000)
86  {
87  ir |= 0x7c00; /* infinity */
88  }
89  else
90  {
91  ir = 0x7fff; /* canonical NaN */
92  }
93  }
94  else if ((ia & 0x7f800000) >= 0x33000000)
95  {
96  int32_t shift = (int32_t) ((ia >> 23) & 0xff) - 127;
97  if (shift > 15)
98  {
99  ir |= 0x7c00; /* infinity */
100  }
101  else
102  {
103  ia = (ia & 0x007fffff) | 0x00800000; /* extract mantissa */
104  if (shift < -14)
105  { /* denormal */
106  ir |= ia >> (-1 - shift);
107  ia = ia << (32 - (-1 - shift));
108  }
109  else
110  { /* normal */
111  ir |= ia >> (24 - 11);
112  ia = ia << (32 - (24 - 11));
113  ir = ir + ((14 + shift) << 10);
114  }
115  /* IEEE-754 round to nearest of even */
116  if ((ia > 0x80000000) || ((ia == 0x80000000) && (ir & 1)))
117  {
118  ir++;
119  }
120  }
121  }
122 
123  this->__x = ir;
124  }

134  __host__ __device__ __forceinline__
135  operator float() const
136  {
137  // Stolen from Andrew Kerr
138 
139  int sign = ((this->__x >> 15) & 1);
140  int exp = ((this->__x >> 10) & 0x1f);
141  int mantissa = (this->__x & 0x3ff);
142  uint32_t f = 0;
143 
144  if (exp > 0 && exp < 31)
145  {
146  // normal
147  exp += 112;
148  f = (sign << 31) | (exp << 23) | (mantissa << 13);
149  }
150  else if (exp == 0)
151  {
152  if (mantissa)
153  {
154  // subnormal
155  exp += 113;
156  while ((mantissa & (1 << 10)) == 0)
157  {
158  mantissa <<= 1;
159  exp--;
160  }
161  mantissa &= 0x3ff;
162  f = (sign << 31) | (exp << 23) | (mantissa << 13);
163  }
164  else if (sign)
165  {
166  f = 0x80000000; // negative zero
167  }
168  else
169  {
170  f = 0x0; // zero
171  }
172  }
173  else if (exp == 31)
174  {
175  if (mantissa)
176  {
177  f = 0x7fffffff; // not a number
178  }
179  else
180  {
181  f = (0xff << 23) | (sign << 31); // inf
182  }
183  }
184  return *reinterpret_cast<float const *>(&f);
185  }

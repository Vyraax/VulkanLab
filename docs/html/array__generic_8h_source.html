424  template <typename T, size_t... Is>
425  ENOKI_INLINE void assign_(T&& value, std::index_sequence<Is...>) {
426  if constexpr (std::is_same_v<array_shape_t<T>, array_shape_t<Derived>> &&
427  std::is_same_v<Value, half>) {
428  #if defined(ENOKI_X86_F16C)
429  using Value2 = value_t<T>;
430 
431  if constexpr (std::is_same_v<Value2, double>) {
432  derived() = float32_array_t<T, false>(value);
433  return;
434  } else if constexpr (std::is_same_v<Value2, float>) {
435  if constexpr (Size == 4) {
436  long long result = detail::mm_cvtsi128_si64(_mm_cvtps_ph(
437  value.derived().m, _MM_FROUND_CUR_DIRECTION));
438  memcpy(m_data.data(), &result, sizeof(long long));
439  return;
440  } else if constexpr (Size == 8) {
441  __m128i result = _mm256_cvtps_ph(value.derived().m,
442  _MM_FROUND_CUR_DIRECTION);
443  _mm_storeu_si128((__m128i *) m_data.data(), result);
444  return;
445  }
446  #if defined(ENOKI_X86_AVX512F)
447  if constexpr (Size == 16) {
448  __m256i result = _mm512_cvtps_ph(value.derived().m,
449  _MM_FROUND_CUR_DIRECTION);
450  _mm256_storeu_si256((__m256i *) m_data.data(), result);
451  return;
452  }
453  #endif
454  }
455  #endif
456  }

478  template <typename T, size_t... Is>
479  ENOKI_INLINE void assign_(T&& value, detail::reinterpret_flag, std::index_sequence<Is...>) {
480  if constexpr (std::is_same_v<array_shape_t<T>, array_shape_t<Derived>> &&
481  std::is_same_v<Value, bool> && detail::has_bitmask_v<T>) {
482  #if defined(ENOKI_X86_AVX512VL)
483  if constexpr (Size == 16) {
484  _mm_storeu_si128((__m128i *) data(),
485  _mm_maskz_set1_epi8((__mmask16) value.bitmask_(), (char) 1));
486  return;
487  } else if constexpr (Size == 8) {
488  uint64_t result = (uint64_t) detail::mm_cvtsi128_si64(
489  _mm_maskz_set1_epi8((__mmask8) value.bitmask_(), (char) 1));
490  memcpy(data(), &result, sizeof(uint64_t));
491  return;
492  } else if constexpr (Size == 4) {
493  uint32_t result = (uint32_t) _mm_cvtsi128_si32(
494  _mm_maskz_set1_epi8((__mmask8) value.bitmask_(), (char) 1));
495  memcpy(data(), &result, sizeof(uint32_t));
496  return;
497  }
498  #elif defined(ENOKI_X86_AVX2) && defined(ENOKI_X86_64)
499  uint32_t k = value.bitmask_();
500  if constexpr (Size == 16) {
501  uint64_t low = (uint64_t) _pdep_u64(k, 0x0101010101010101ull);
502  uint64_t hi = (uint64_t) _pdep_u64(k >> 8, 0x0101010101010101ull);
503  memcpy((uint8_t *) data(), &low, sizeof(uint64_t));
504  memcpy((uint8_t *) data() + sizeof(uint64_t), &hi, sizeof(uint64_t));
505  return;
506  } else if constexpr (Size == 8) {
507  uint64_t result = (uint64_t) _pdep_u64(k, 0x0101010101010101ull);
508  memcpy(data(), &result, sizeof(uint64_t));
509  return;
510  } else if constexpr (Size == 4) {
511  uint32_t result = (uint32_t) _pdep_u32(k, 0x01010101ull);
512  memcpy(data(), &result, sizeof(uint32_t));
513  return;
514  }
515  #endif
516  }

63  ENOKI_REINTERPRET_KMASK(bool) {
64  __m128i value;
65  if constexpr (Size == 16)
66  value = _mm_loadu_si128((__m128i *) a.derived().data());
67  else if constexpr (Size == 8)
68  value = _mm_loadl_epi64((const __m128i *) a.derived().data());
69  else if constexpr (Size == 4 || Size == 3)
70  value = _mm_cvtsi32_si128(*((const int *) a.derived().data()));
71  else if constexpr (Size == 2)
72  value = _mm_cvtsi32_si128((int) *((const short *) a.derived().data()));
73  else
74  static_assert(detail::false_v<Value2>, "Unsupported number of elements");
75 
76  #if defined(ENOKI_X86_AVX512VL) && defined(ENOKI_X86_AVX512BW)
77  k = (Register) _mm_test_epi8_mask(value, _mm_set1_epi8((char) 0xFF));
78  #else
79  k = (Register) _mm512_test_epi32_mask(_mm512_cvtepi8_epi32(value),
80  _mm512_set1_epi8((char) 0xFF));
81  #endif
82  }

157  ENOKI_INLINE bool all_() const {
158  if constexpr (Size_ == 16)
159  return _mm512_kortestc(k, k);
160  else if constexpr (Size_ == 8)
161  return k == BitMask;
162  else
163  return (k & BitMask) == BitMask;

166  ENOKI_INLINE bool any_() const {
